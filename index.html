// コイン表示部分のレンダリング function renderCoinDisplay() { const account = appData.accounts[currentAccountIndex]; const currentChallenge = account.challenges[currentChallengeIndex]; // チャレンジが設定されていない場合は非表示 if (currentChallengeIndex < 0 || !currentChallenge) { $('currentCoins').textContent = '—'; $('currentCoins').classList.add('disabled'); $('currentValue').textContent = '—'; $('coinValueLabel').textContent = 'チャレンジを設定してください'; $('removeCoinBtn').disabled = true; $('addCoinBtn').disabled = true; $('exchangeInput').disabled = true; $('exchangeBtn').disabled = true; $('totalCoins').textContent = '—'; $('totalCoinsLabel').textContent = '総獲得コイン'; return; } $('currentCoins').classList.remove('disabled'); $('removeCoinBtn').disabled = currentChallenge.coins <= 0; $('addCoinBtn').disabled = false; $('currentCoins').textContent = currentChallenge.coins; $('currentValue').textContent = currentChallenge.coins * currentChallenge.value; // ★★★ 修正箇所 ★★★ // チャレンジ名 (1コイン = XX円) からチャレンジ名を削除し、金額表示のみにする $('coinValueLabel').innerHTML = `1コイン = ${currentChallenge.value}円`; // ★★★ 修正箇所 ここまで ★★★ // 合計コイン表示 const totalCoins = account.challenges.reduce((sum, chal) => sum + chal.totalCoins, 0); $('totalCoins').textContent = totalCoins; $('totalCoinsLabel').textContent = 'これまでのコイン合計'; // 交換ボタンの制御 const exchangeInput = $('exchangeInput'); const exchangeBtn = $('exchangeBtn'); const exchangeAmount = parseInt(exchangeInput.value); // exchangeInputの最大値と最小値を設定 exchangeInput.min = 1; exchangeInput.max = currentChallenge.coins; if (isNaN(exchangeAmount) || exchangeAmount < 1 || exchangeAmount > currentChallenge.coins) { exchangeBtn.disabled = true; } else { exchangeBtn.disabled = false; } }
